/**@class android.telephony.ims.ImsCallSessionListener
@extends java.lang.Object

 Listener interface for notifying the Framework's {@link android.telephony.ims.ImsCallSession} for updates to an ongoing
 IMS call.

 @hide
*/
var ImsCallSessionListener = {

/**A request has been sent out to initiate a new IMS call session and a 1xx response has been
 received from the network.
*/
callSessionProgressing : function(  ) {},

/**The IMS call session has been initiated.
@param {Object {ImsCallProfile}} profile the associated {@link ImsCallProfile}.
*/
callSessionInitiated : function(  ) {},

/**The IMS call session establishment has failed.
@param {Object {ImsReasonInfo}} reasonInfo {@link ImsReasonInfo} detailing the reason of the IMS call session
 establishment failure.
*/
callSessionInitiatedFailed : function(  ) {},

/**The IMS call session has been terminated.
@param {Object {ImsReasonInfo}} reasonInfo {@link ImsReasonInfo} detailing the reason of the session termination.
*/
callSessionTerminated : function(  ) {},

/**The IMS call session has started the process of holding the call. If it fails,
 {@link #callSessionHoldFailed}(ImsReasonInfo) should be called.

 If the IMS call session is resumed, call {@link #callSessionResumed}(ImsCallProfile).
@param {Object {ImsCallProfile}} profile The associated {@link ImsCallProfile} of the call session that has been put
 on hold.
*/
callSessionHeld : function(  ) {},

/**The IMS call session has failed to be held.
@param {Object {ImsReasonInfo}} reasonInfo {@link ImsReasonInfo} detailing the reason of the session hold failure.
*/
callSessionHoldFailed : function(  ) {},

/**This IMS Call session has been put on hold by the remote party.
@param {Object {ImsCallProfile}} profile The {@link ImsCallProfile} associated with this IMS call session.
*/
callSessionHoldReceived : function(  ) {},

/**The IMS call session has started the process of resuming the call. If the process of resuming
 the call fails, call {@link #callSessionResumeFailed}(ImsReasonInfo).
@param {Object {ImsCallProfile}} profile The {@link ImsCallProfile} associated with this IMS call session.
*/
callSessionResumed : function(  ) {},

/**The IMS call session resume has failed.
@param {Object {ImsReasonInfo}} reasonInfo {@link ImsReasonInfo} containing the detailed reason of the session resume
 failure.
*/
callSessionResumeFailed : function(  ) {},

/**The remote party has resumed this IMS call session.
@param {Object {ImsCallProfile}} profile {@link ImsCallProfile} associated with the IMS call session.
*/
callSessionResumeReceived : function(  ) {},

/**The IMS call session merge has been started.  At this point, the {@code newSession}
 represents the IMS call session which represents the new merged conference and has been
 initiated to the IMS conference server.
@param {Object {ImsCallSessionImplBase}} newSession the {@link ImsCallSessionImplBase} that represents the merged active & held
 sessions.
@param {Object {ImsCallProfile}} profile The {@link ImsCallProfile} associated with this IMS call session.
*/
callSessionMergeStarted : function(  ) {},

/**Compatibility method for older implementations.
 See {@link #callSessionMergeStarted(ImsCallSessionImplBase, android.telephony.ims.ImsCallProfile)}.
@hide 
*/
callSessionMergeStarted : function(  ) {},

/**The session merge is successful and the merged {@link android.telephony.ims.ImsCallSession} is active.
@param {Object {ImsCallSessionImplBase}} newSession the new {@link ImsCallSessionImplBase}
                  that represents the conference IMS call
 session.
*/
callSessionMergeComplete : function(  ) {},

/**Compatibility method for older implementations of ImsService.

 See {@link #callSessionMergeComplete}(ImsCallSessionImplBase)}.
@hide 
*/
callSessionMergeComplete : function(  ) {},

/**The IMS call session merge has failed.
@param {Object {ImsReasonInfo}} reasonInfo {@link ImsReasonInfo} contining the reason for the call merge failure.
*/
callSessionMergeFailed : function(  ) {},

/**The IMS call session profile has been updated. Does not include holding or resuming a call.
@param {Object {ImsCallProfile}} profile The {@link ImsCallProfile} associated with the updated IMS call session.
*/
callSessionUpdated : function(  ) {},

/**The IMS call session profile update has failed.
@param {Object {ImsReasonInfo}} reasonInfo {@link ImsReasonInfo} containing a reason for the session update failure.
*/
callSessionUpdateFailed : function(  ) {},

/**The IMS call session profile has received an update from the remote user.
@param {Object {ImsCallProfile}} profile The new {@link ImsCallProfile} associated with the update.
*/
callSessionUpdateReceived : function(  ) {},

/**Called when the session has been extended to a conference session.

 If the conference extension fails, call
 {@link #callSessionConferenceExtendFailed}(ImsReasonInfo).
@param {Object {ImsCallSessionImplBase}} newSession the session object that is extended to the conference from the active
 IMS Call session.
@param {Object {ImsCallProfile}} profile The {@link ImsCallProfile} associated with the IMS call session.
*/
callSessionConferenceExtended : function(  ) {},

/**Compatibility method to interface with older versions of ImsService.
 See {@link #callSessionConferenceExtended(ImsCallSessionImplBase, android.telephony.ims.ImsCallProfile)}.
@hide 
*/
callSessionConferenceExtended : function(  ) {},

/**The previous conference extension has failed.
@param {Object {ImsReasonInfo}} reasonInfo {@link ImsReasonInfo} containing the detailed reason of the conference
 extension failure.
*/
callSessionConferenceExtendFailed : function(  ) {},

/**A conference extension has been received received from the remote party.
@param {Object {ImsCallSessionImplBase}} newSession An {@link ImsCallSessionImplBase}
                   representing the extended IMS call session.
@param {Object {ImsCallProfile}} profile The {@link ImsCallProfile} associated with the new IMS call session.
*/
callSessionConferenceExtendReceived : function(  ) {},

/**Compatibility method to interface with older versions of ImsService.
 See {@link #callSessionConferenceExtendReceived(ImsCallSessionImplBase, android.telephony.ims.ImsCallProfile)}.
@hide 
*/
callSessionConferenceExtendReceived : function(  ) {},

/**The request to invite participants to the conference has been delivered to the conference
 server.
*/
callSessionInviteParticipantsRequestDelivered : function(  ) {},

/**The previous request to invite participants to the conference (see
 {@link #callSessionInviteParticipantsRequestDelivered}()) has failed.
@param {Object {ImsReasonInfo}} reasonInfo {@link ImsReasonInfo} detailing the reason forthe conference invitation
 failure.
*/
callSessionInviteParticipantsRequestFailed : function(  ) {},

/**The request to remove participants from the conference has been delivered to the conference
 server.
*/
callSessionRemoveParticipantsRequestDelivered : function(  ) {},

/**The previous request to remove participants from the conference (see
 {@link #callSessionRemoveParticipantsRequestDelivered}()) has failed.
@param {Object {ImsReasonInfo}} reasonInfo {@link ImsReasonInfo} detailing the reason for the conference removal
 failure.
*/
callSessionRemoveParticipantsRequestFailed : function(  ) {},

/**The IMS call session's conference state has changed.
@param {Object {ImsConferenceState}} state The new {@link ImsConferenceState} associated with the conference.
*/
callSessionConferenceStateUpdated : function(  ) {},

/**The IMS call session has received a Ussd message.
@param {Number} mode The mode of the USSD message, either
 {@link ImsCallSessionImplBase#USSD_MODE_NOTIFY} or
 {@link ImsCallSessionImplBase#USSD_MODE_REQUEST}.
@param {String} ussdMessage The USSD message.
*/
callSessionUssdMessageReceived : function(  ) {},

/**An {@link android.telephony.ims.ImsCallSession} may potentially handover from one radio
 technology to another.
@param {Number} srcAccessTech The source radio access technology; one of the access technology
 constants defined in {@link android.telephony.ServiceState}. For example
 {@link android.telephony.ServiceState#RIL_RADIO_TECHNOLOGY_LTE}.
@param {Number} targetAccessTech The target radio access technology; one of the access technology
 constants defined in {@link android.telephony.ServiceState}. For example
 {@link android.telephony.ServiceState#RIL_RADIO_TECHNOLOGY_LTE}.
*/
callSessionMayHandover : function(  ) {},

/**The IMS call session's access technology has changed.
@param {Number} srcAccessTech original access technology, defined in
 {@link android.telephony.ServiceState}.
@param {Number} targetAccessTech new access technology, defined in
 {@link android.telephony.ServiceState}.
@param {Object {ImsReasonInfo}} reasonInfo The {@link ImsReasonInfo} associated with this handover.
*/
callSessionHandover : function(  ) {},

/**The IMS call session's access technology change has failed..
@param {Number} srcAccessTech original access technology
@param {Number} targetAccessTech new access technology
@param {Object {ImsReasonInfo}} reasonInfo An {@link ImsReasonInfo} detailing the reason for the failure.
*/
callSessionHandoverFailed : function(  ) {},

/**The TTY mode has been changed by the remote party.
@param {Number} mode one of the following: -
             {@link com.android.internal.telephony.Phone#TTY_MODE_OFF} -
             {@link com.android.internal.telephony.Phone#TTY_MODE_FULL} -
             {@link com.android.internal.telephony.Phone#TTY_MODE_HCO} -
             {@link com.android.internal.telephony.Phone#TTY_MODE_VCO}
*/
callSessionTtyModeReceived : function(  ) {},

/**The multiparty state has been changed for this {@code ImsCallSession}.
@param {Boolean} isMultiParty {@code true} if the session became multiparty, {@code false} otherwise.
*/
callSessionMultipartyStateChanged : function(  ) {},

/**Supplementary service information has been received for the current IMS call session.
@param {Object {ImsSuppServiceNotification}} suppSrvNotification The {@link ImsSuppServiceNotification} containing the change.
*/
callSessionSuppServiceReceived : function(  ) {},

/**An RTT modify request has been received from the remote party.
@param {Object {ImsCallProfile}} callProfile An {@link ImsCallProfile} with the updated attributes
*/
callSessionRttModifyRequestReceived : function(  ) {},

/**An RTT modify response has been received.
@param {Number} status the received response for RTT modify request.
*/
callSessionRttModifyResponseReceived : function(  ) {},

/**An RTT message has been received from the remote party.
@param {String} rttMessage The RTT message that has been received.
*/
callSessionRttMessageReceived : function(  ) {},

/**While in call, there has been a change in RTT audio indicator.
@param {Object {ImsStreamMediaProfile}} profile updated ImsStreamMediaProfile
*/
callSessionRttAudioIndicatorChanged : function(  ) {},

/**The call quality has changed.
@param {Object {CallQuality}} callQuality The new call quality
*/
callQualityChanged : function(  ) {},


};
