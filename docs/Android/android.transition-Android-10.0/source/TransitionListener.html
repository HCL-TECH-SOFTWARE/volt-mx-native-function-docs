/**@class android.transition.Transition.TransitionListener
 A transition listener receives notifications from a transition.
 Notifications indicate transition lifecycle events.
*/
var TransitionListener = {

/**Notification about the start of the transition.
@param {Object {Transition}} transition The started transition.
*/
onTransitionStart : function(  ) {},

/**Notification about the end of the transition. Canceled transitions

 through to completion.
@param {Object {Transition}} transition The transition which reached its end.
*/
onTransitionEnd : function(  ) {},

@param {Object {Transition}} transition The transition which was canceled.
*/

 {@link #createAnimator(android.view.ViewGroup, android.transition.TransitionValues, android.transition.TransitionValues)
 createAnimator()} time.
@param {Object {Transition}} transition The transition which was paused.
*/
onTransitionPause : function(  ) {},

@param {Object {Transition}} transition The transition which was resumed.
