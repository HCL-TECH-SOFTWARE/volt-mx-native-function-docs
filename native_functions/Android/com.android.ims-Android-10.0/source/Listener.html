/**@class com.android.ims.ImsCall.Listener
@extends java.lang.Object

 Listener for events relating to an IMS call, such as when a call is being
 received ("on ringing") or a call is outgoing ("on calling").
 <p>Many of these events are also received by {@link com.android.ims.ImsCallSession.Listener}.</p>
*/
var Listener = {

/**Called when a request is sent out to initiate a new call
 and 1xx response is received from the network.
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallProgressing : function(  ) {},

/**Called when the call is established.
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallStarted : function(  ) {},

/**Called when the call setup is failed.
 The default implementation calls {@link #onCallError}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the call setup failure
*/
onCallStartFailed : function(  ) {},

/**Called when the call is terminated.
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the call termination
*/
onCallTerminated : function(  ) {},

/**Called when the call is in hold.
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallHeld : function(  ) {},

/**Called when the call hold is failed.
 The default implementation calls {@link #onCallError}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the call hold failure
*/
onCallHoldFailed : function(  ) {},

/**Called when the call hold is received from the remote user.
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallHoldReceived : function(  ) {},

/**Called when the call is in call.
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallResumed : function(  ) {},

/**Called when the call resume is failed.
 The default implementation calls {@link #onCallError}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the call resume failure
*/
onCallResumeFailed : function(  ) {},

/**Called when the call resume is received from the remote user.
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallResumeReceived : function(  ) {},

/**Called when the call is in call.
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the active IMS call
@param {Object {ImsCall}} peerCall the call object that carries out the held IMS call
@param {Boolean} swapCalls {@code true} if the foreground and background calls should be swapped
                              now that the merge has completed.
*/
onCallMerged : function(  ) {},

/**Called when the call merge is failed.
 The default implementation calls {@link #onCallError}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the call merge failure
*/
onCallMergeFailed : function(  ) {},

/**Called when the call is updated (except for hold/unhold).
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallUpdated : function(  ) {},

/**Called when the call update is failed.
 The default implementation calls {@link #onCallError}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the call update failure
*/
onCallUpdateFailed : function(  ) {},

/**Called when the call update is received from the remote user.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallUpdateReceived : function(  ) {},

/**Called when the call is extended to the conference call.
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsCall}} newCall the call object that is extended to the conference from the active call
*/
onCallConferenceExtended : function(  ) {},

/**Called when the conference extension is failed.
 The default implementation calls {@link #onCallError}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the conference extension failure
*/
onCallConferenceExtendFailed : function(  ) {},

/**Called when the conference extension is received from the remote user.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsCall}} newCall the call object that is extended to the conference from the active call
*/
onCallConferenceExtendReceived : function(  ) {},

/**Called when the invitation request of the participants is delivered to
 the conference server.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallInviteParticipantsRequestDelivered : function(  ) {},

/**Called when the invitation request of the participants is failed.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the conference invitation failure
*/
onCallInviteParticipantsRequestFailed : function(  ) {},

/**Called when the removal request of the participants is delivered to
 the conference server.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallRemoveParticipantsRequestDelivered : function(  ) {},

/**Called when the removal request of the participants is failed.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the conference removal failure
*/
onCallRemoveParticipantsRequestFailed : function(  ) {},

/**Called when the conference state is updated.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsConferenceState}} state state of the participant who is participated in the conference call
*/
onCallConferenceStateUpdated : function(  ) {},

/**Called when the state of IMS conference participant(s) has changed.
@param {Object {ImsCall}} call the call object that carries out the IMS call.
@param {Object {java.util.List}} participants the participant(s) and their new state information.
*/
onConferenceParticipantsStateChanged : function(  ) {},

/**Called when the USSD message is received from the network.
@param {Object {ImsCall}} mode mode of the USSD message (REQUEST / NOTIFY)
@param {Number} ussdMessage USSD message
*/
onCallUssdMessageReceived : function(  ) {},

/**Called when an error occurs. The default implementation is no op.
 overridden. The default implementation is no op. Error events are
 not re-directed to this callback and are handled in {@link #onCallError}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of this error
@see ImsReasonInfo
*/
onCallError : function(  ) {},

/**Called when an event occurs and the corresponding callback is not
 overridden. The default implementation is no op. Error events are
 not re-directed to this callback and are handled in {@link #onCallError}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallStateChanged : function(  ) {},

/**Called when the call moves the hold state to the conversation state.
 For example, when merging the active & hold call, the state of all the hold call
 will be changed from hold state to conversation state.
 This callback method can be invoked even though the application does not trigger
 any operations.
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Number} state the detailed state of call state changes;
      Refer to CALL_STATE_* in {@link ImsCall}
*/
onCallStateChanged : function(  ) {},

/**Called when the call supp service is received
 The default implementation calls {@link #onCallStateChanged}.
@param {Object {ImsCall}} call the call object that carries out the IMS call
*/
onCallSuppServiceReceived : function(  ) {},

/**Called when TTY mode of remote party changed
@param {Object {ImsCall}} call the call object that carries out the IMS call
@param {Number} mode TTY mode of remote party
*/
onCallSessionTtyModeReceived : function(  ) {},

/**Called when handover occurs from one access technology to another.
@param {Object {ImsCall}} imsCall ImsCall object
@param {Number} srcAccessTech original access technology
@param {Number} targetAccessTech new access technology
@param {Object {ImsReasonInfo}} reasonInfo
*/
onCallHandover : function(  ) {},

/**Called when the remote party issues an RTT modify request
@param {Object {ImsCall}} imsCall ImsCall object
*/
onRttModifyRequestReceived : function(  ) {},

/**Called when the remote party responds to a locally-issued RTT request.
@param {Object {ImsCall}} imsCall ImsCall object
@param {Number} status The status of the request. See
               {@link Connection.RttModifyStatus} for possible values.
*/
onRttModifyResponseReceived : function(  ) {},

/**Called when the remote party has sent some characters via RTT
@param {Object {ImsCall}} imsCall ImsCall object
@param {String} message A string containing the transmitted characters.
*/
onRttMessageReceived : function(  ) {},

/**Called when handover from one access technology to another fails.
@param {Object {ImsCall}} imsCall call that failed the handover.
@param {Number} srcAccessTech original access technology
@param {Number} targetAccessTech new access technology
@param {Object {ImsReasonInfo}} reasonInfo
*/
onCallHandoverFailed : function(  ) {},

/**Notifies of a change to the multiparty state for this {@code ImsCall}.
@param {Object {ImsCall}} imsCall The IMS call.
@param {Boolean} isMultiParty {@code true} if the call became multiparty, {@code false}
      otherwise.
*/
onMultipartyStateChanged : function(  ) {},

/**Called when rtt call audio indicator has changed.
@param {Object {ImsCall}} imsCall ImsCall object
@param {Object {ImsStreamMediaProfile}} profile updated ImsStreamMediaProfile profile.
*/
onRttAudioIndicatorChanged : function(  ) {},

/**Called when the call quality has changed.
@param {Object {ImsCall}} imsCall ImsCall object
@param {Object {CallQuality}} callQuality the updated CallQuality
*/
onCallQualityChanged : function(  ) {},


};
