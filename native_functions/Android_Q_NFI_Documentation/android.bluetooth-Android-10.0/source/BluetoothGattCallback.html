var BluetoothGattCallback = {

@param {Number} rxPhy the receiver PHY in use. One of {@link BluetoothDevice#PHY_LE_1M}, {@link
 BluetoothDevice#PHY_LE_2M}, and {@link BluetoothDevice#PHY_LE_CODED}.

 operation succeeds.
*/
onPhyUpdate : function(  ) {},

/**Callback triggered as result of {@link android.bluetooth.BluetoothGatt#readPhy}
@param {Object {BluetoothGatt}} gatt GATT client
@param {Number} txPhy the transmitter PHY in use. One of {@link BluetoothDevice#PHY_LE_1M}, {@link
 BluetoothDevice#PHY_LE_2M}, and {@link BluetoothDevice#PHY_LE_CODED}.
@param {Number} rxPhy the receiver PHY in use. One of {@link BluetoothDevice#PHY_LE_1M}, {@link
 BluetoothDevice#PHY_LE_2M}, and {@link BluetoothDevice#PHY_LE_CODED}.
@param {Number} status Status of the PHY read operation. {@link BluetoothGatt#GATT_SUCCESS} if the
 operation succeeds.
*/
onPhyRead : function(  ) {},

/**Callback indicating when GATT client has connected/disconnected to/from a remote
 GATT server.
@param {Object {BluetoothGatt}} gatt GATT client
@param {Number} status Status of the connect or disconnect operation. {@link
 BluetoothGatt#GATT_SUCCESS} if the operation succeeds.
@param {Number} newState Returns the new connection state. Can be one of {@link
 BluetoothProfile#STATE_DISCONNECTED} or {@link BluetoothProfile#STATE_CONNECTED}
*/
onConnectionStateChange : function(  ) {},

/**Callback invoked when the list of remote services, characteristics and descriptors
 for the remote device have been updated, ie new services have been discovered.
@param {Object {BluetoothGatt}} gatt GATT client invoked {@link BluetoothGatt#discoverServices}
@param {Number} status {@link BluetoothGatt#GATT_SUCCESS} if the remote device has been explored
 successfully.
*/
onServicesDiscovered : function(  ) {},

/**Callback reporting the result of a characteristic read operation.
@param {Object {BluetoothGatt}} gatt GATT client invoked {@link BluetoothGatt#readCharacteristic}
@param {Object {BluetoothGattCharacteristic}} characteristic Characteristic that was read from the associated remote device.
@param {Number} status {@link BluetoothGatt#GATT_SUCCESS} if the read operation was completed
 successfully.
*/
onCharacteristicRead : function(  ) {},

/**Callback indicating the result of a characteristic write operation.

 <p>If this callback is invoked while a reliable write transaction is

 the application must abort the reliable write transaction.
@param {Object {BluetoothGatt}} gatt GATT client invoked {@link BluetoothGatt#writeCharacteristic}
@param {Object {BluetoothGattCharacteristic}} characteristic Characteristic that was written to the associated remote device.
@param {Number} status The result of the write operation {@link BluetoothGatt#GATT_SUCCESS} if the
 operation succeeds.
*/
onCharacteristicWrite : function(  ) {},

/**Callback triggered as a result of a remote characteristic notification.
@param {Object {BluetoothGatt}} gatt GATT client the characteristic is associated with
@param {Object {BluetoothGattCharacteristic}} characteristic Characteristic that has been updated as a result of a remote
 notification event.
*/
onCharacteristicChanged : function(  ) {},

/**Callback reporting the result of a descriptor read operation.
@param {Object {BluetoothGatt}} gatt GATT client invoked {@link BluetoothGatt#readDescriptor}
@param {Object {BluetoothGattDescriptor}} descriptor Descriptor that was read from the associated remote device.
@param {Number} status {@link BluetoothGatt#GATT_SUCCESS} if the read operation was completed
 successfully
*/
onDescriptorRead : function(  ) {},

/**Callback indicating the result of a descriptor write operation.
@param {Object {BluetoothGatt}} gatt GATT client invoked {@link BluetoothGatt#writeDescriptor}
@param {Object {BluetoothGattDescriptor}} descriptor Descriptor that was writte to the associated remote device.
@param {Number} status The result of the write operation {@link BluetoothGatt#GATT_SUCCESS} if the
 operation succeeds.
*/
onDescriptorWrite : function(  ) {},

/**Callback invoked when a reliable write transaction has been completed.
@param {Object {BluetoothGatt}} gatt GATT client invoked {@link BluetoothGatt#executeReliableWrite}
@param {Number} status {@link BluetoothGatt#GATT_SUCCESS} if the reliable write transaction was
 executed successfully
*/
onReliableWriteCompleted : function(  ) {},

/**Callback reporting the RSSI for a remote device connection.

 This callback is triggered in response to the
 {@link android.bluetooth.BluetoothGatt#readRemoteRssi} function.
@param {Object {BluetoothGatt}} gatt GATT client invoked {@link BluetoothGatt#readRemoteRssi}
@param {Number} rssi The RSSI value for the remote device
@param {Number} status {@link BluetoothGatt#GATT_SUCCESS} if the RSSI was read successfully
*/
onReadRemoteRssi : function(  ) {},

/**Callback indicating the MTU for a given device connection has changed.

 This callback is triggered in response to the
 {@link android.bluetooth.BluetoothGatt#requestMtu} function, or in response to a connection
 event.
@param {Object {BluetoothGatt}} gatt GATT client invoked {@link BluetoothGatt#requestMtu}
@param {Number} mtu The new MTU size
@param {Number} status {@link BluetoothGatt#GATT_SUCCESS} if the MTU has been changed successfully
*/
onMtuChanged : function(  ) {},

/**Callback indicating the connection parameters were updated.
@param {Object {BluetoothGatt}} gatt GATT client involved

@param {Number} latency Slave latency for the connection in number of connection events. Valid range

@param {Number} timeout Supervision timeout for this connection, in 10ms unit. Valid range is from 10
 (0.1s) to 3200 (32s)
@param {Number} status {@link BluetoothGatt#GATT_SUCCESS} if the connection has been updated
 successfully
@hide 
*/
