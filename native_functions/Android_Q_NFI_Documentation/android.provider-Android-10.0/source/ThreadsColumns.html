/**@class android.provider.Telephony.ThreadsColumns
 implements android.provider.BaseColumns

 Columns for the "threads" table used by MMS and SMS.
*/
var ThreadsColumns = {

/** The date at which the thread was created.
 <P>Type: INTEGER (long)</P>
*/
DATE : "date",
/** A string encoding of the recipient IDs of the recipients of
 the message, in numerical order and separated by spaces.
 <P>Type: TEXT</P>
*/
RECIPIENT_IDS : "recipient_ids",
/** The message count of the thread.
 <P>Type: INTEGER</P>
*/
MESSAGE_COUNT : "message_count",
/** Indicates whether all messages of the thread have been read.
 <P>Type: INTEGER</P>
*/
READ : "read",
/** The snippet of the latest message in the thread.
 <P>Type: TEXT</P>
*/
SNIPPET : "snippet",
/** The charset of the snippet.
 <P>Type: INTEGER</P>
*/
SNIPPET_CHARSET : "snippet_cs",
/** Type of the thread, either {@link android.provider.Telephony.Threads#COMMON_THREAD} or
 {@link android.provider.Telephony.Threads#BROADCAST_THREAD}.
 <P>Type: INTEGER</P>
*/
TYPE : "type",
/** Indicates whether there is a transmission error in the thread.
 <P>Type: INTEGER</P>
*/
ERROR : "error",
/** Indicates whether this thread contains any attachments.
 <P>Type: INTEGER</P>
*/
HAS_ATTACHMENT : "has_attachment",
/** If the thread is archived
 <P>Type: INTEGER (boolean)</P>
*/
ARCHIVED : "archived",

};
